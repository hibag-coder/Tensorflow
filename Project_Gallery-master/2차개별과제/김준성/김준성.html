from keras.datasets import mnist
from keras.utils import np_utils
from keras.models import Sequential
from keras import regularizers
from keras.layers import Activation, Conv2D, Dense, Dropout, Flatten, MaxPooling2D
from keras.callbacks import ModelCheckpoint,EarlyStopping
from keras.regularizers import l2

# Sequential Convolutional Neural Network
model = Sequential()

# model.add(  , ,    ,  )         
model.add(Conv2D(64, (5, 5), padding='same',
                 input_shape=(32, 32, 3)))
# input_shape=(, , (3)  (1))
# kernel_regularizer weight regularization( ) :           (       )
# L2 regularization(=weight decay) :     ( L2 norm)
model.add(Activation('relu'))
model.add(Conv2D(64, kernel_size=(5, 5)))
model.add(Activation('relu'))
model.add(MaxPooling2D(pool_size=2))
model.add(Dropout(0.1)) 

model.add(Conv2D(64, (5, 5), padding='same', activation='relu'))
model.add(Conv2D(64, (5, 5), padding='same', activation='softsign'))
model.add(MaxPooling2D(pool_size=2))
model.add(Dropout(0.25)) #   25%  

model.add(Conv2D(64, (5, 5), padding='same', activation='relu'))
model.add(Conv2D(64, (5, 5), padding='same', activation='softsign'))
model.add(MaxPooling2D(pool_size=2))
model.add(Dropout(0.25))

model.add(Conv2D(64, (5, 5), padding='same', activation='relu'))
model.add(Conv2D(64, (5, 5), padding='same', activation='softsign'))
model.add(MaxPooling2D(pool_size=2))
model.add(Dropout(0.25))

model.add(Flatten()) # CNN MaxPooling    2      1  
model.add(Dense(512,  activation='relu'))
model.add(Dropout(0.5))
model.add(Dense(10, activation='softmax'))
model.summary()
